'0' + horas : horas; minutos = minutos < 10 ? '0' + minutos : minutos; segundos = segundos < 10 ? '0' + segundos : segundos; // Formata a string da hora const horaFormatada = `${horas}:${minutos}:${segundos}`; // Encontra o elemento com o ID 'relogio' e atualiza o texto document.getElementById('relogio').innerText = horaFormatada; } // Chama a função a cada 1000 milissegundos (1 segundo) setInterval(atualizarRelogio, 1000); // Chama a função uma vez para não haver atraso inicial atualizarRelogio();
